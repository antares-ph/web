# ═══════════════════════════════════════════════════════════
#  ANTARES — SEGURIDAD DE SERVIDOR (.htaccess)
#  Compatible con Apache 2.4 / cPanel / Hostinger / SiteGround
# ═══════════════════════════════════════════════════════════

Options -Indexes -ExecCGI
ServerSignature Off

# ─── HEADERS DE SEGURIDAD HTTP ───────────────────────────────
<IfModule mod_headers.c>

  # Bloquear carga en iframes (anti-clickjacking)
  Header always set X-Frame-Options "DENY"

  # Evitar que el navegador "adivine" el tipo MIME
  Header always set X-Content-Type-Options "nosniff"

  # XSS Protection básico (navegadores legacy)
  Header always set X-XSS-Protection "1; mode=block"

  # No enviar Referer a terceros
  Header always set Referrer-Policy "no-referrer"

  # Permissions Policy — deshabilitar APIs innecesarias
  Header always set Permissions-Policy "camera=(), microphone=(), geolocation=(), payment=(), usb=(), interest-cohort=()"

  # Content Security Policy
  # Ajustar si añades más CDNs
  Header always set Content-Security-Policy "\
    default-src 'self'; \
    script-src 'self' https://cdnjs.cloudflare.com https://fonts.googleapis.com 'unsafe-inline'; \
    style-src 'self' https://fonts.googleapis.com 'unsafe-inline'; \
    font-src 'self' https://fonts.gstatic.com; \
    img-src 'self' data: blob:; \
    connect-src 'self'; \
    frame-ancestors 'none'; \
    base-uri 'self'; \
    form-action 'self'; \
    object-src 'none'; \
    media-src 'self'"

  # Forzar HTTPS (descomentar cuando tengas SSL activo)
  # Header always set Strict-Transport-Security "max-age=31536000; includeSubDomains; preload"

  # Eliminar cabeceras que revelan tecnología
  Header always unset X-Powered-By
  Header always unset Server
  Header always unset X-AspNet-Version
  Header always unset X-AspNetMvc-Version

</IfModule>

# ─── ANTI-HOTLINKING DE IMÁGENES ─────────────────────────────
# Bloquea que otros sitios incrusten tus imágenes directamente
<IfModule mod_rewrite.c>
  RewriteEngine On
  RewriteCond %{HTTP_REFERER} !^$
  RewriteCond %{HTTP_REFERER} !^https?://(www\.)?ant4rez\.com [NC]
  RewriteCond %{HTTP_REFERER} !^https?://antares-ph\.github\.io [NC]
  RewriteCond %{HTTP_REFERER} !^https?://antares-bhd\.pages\.dev [NC]
  RewriteCond %{HTTP_REFERER} !^https?://localhost [NC]
  RewriteRule \.(jpg|jpeg|png|gif|webp|svg)$ - [F,NC,L]
</IfModule>

# ─── BLOQUEAR ACCESO A ARCHIVOS SENSIBLES ────────────────────
<FilesMatch "\.(htaccess|htpasswd|env|git|gitignore|DS_Store|log|bak|swp|tmp|sql|conf)$">
  Order Allow,Deny
  Deny from all
</FilesMatch>

# ─── BLOQUEAR BOTS CONOCIDOS Y SCRAPERS ──────────────────────
<IfModule mod_rewrite.c>
  RewriteCond %{HTTP_USER_AGENT} (wget|curl|python-requests|scrapy|httrack|webzip|webcopier|teleport|webstripper|webwhacker|larbin|libwww|lwp-trivial|webwalk|libwww-perl|go-http-client) [NC]
  RewriteRule .* - [F,L]
</IfModule>

# ─── PROTECCIÓN CONTRA MÉTODOS HTTP PELIGROSOS ───────────────
<LimitExcept GET POST HEAD>
  Order Allow,Deny
  Deny from all
</LimitExcept>

# ─── CACHE PARA PERFORMANCE ──────────────────────────────────
<IfModule mod_expires.c>
  ExpiresActive On
  ExpiresByType image/jpg "access plus 1 year"
  ExpiresByType image/jpeg "access plus 1 year"
  ExpiresByType image/png "access plus 1 year"
  ExpiresByType image/webp "access plus 1 year"
  ExpiresByType image/svg+xml "access plus 1 month"
  ExpiresByType text/css "access plus 1 month"
  ExpiresByType application/javascript "access plus 1 month"
  ExpiresByType font/woff2 "access plus 1 year"
</IfModule>

# ─── COMPRESIÓN GZIP ─────────────────────────────────────────
<IfModule mod_deflate.c>
  AddOutputFilterByType DEFLATE text/html text/css application/javascript application/json image/svg+xml
</IfModule>
